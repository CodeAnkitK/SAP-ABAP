# Complete SAP ABAP Employee Management Learning Project

## Project Overview: "ZEMP_MGMT" - Enterprise Employee System

This **comprehensive learning project** combines **Infotypes**, **Dynpro screens**, **Menu Painter**, **BDC**, **BAPI**, and **Message classes** into one professional employee management system. Students will build a transaction `ZEMP_MGMT` that mirrors real SAP HR functionality.

***

## 🎯 **Learning Objectives**

By project completion, students will master:

```
✅ Dynpro screen navigation (4 screens)
✅ Menu Painter with application toolbar
✅ PA40-style BDC for employee creation
✅ BAPI_EMPLOYEE_GETDATA integration  
✅ Custom table + Infotype simulation
✅ Message class with all message types
✅ Error handling + validation
✅ ALV reporting for employee data
```

***

## 1. **Custom Table Creation** (SE11)

### Table: `ZEMP_MASTER`

| Field | Type | Key | Description |
|-------|------|-----|-------------|
| MANDT | MANDT | X | Client |
| PERNR | NUMC 8 | X | Personnel Number |
| EMP_NAME | CHAR 40 | | Employee Name |
| DEPT | CHAR 20 | | Department |
| DESIGNATION | CHAR 30 | | Position |
| BASIC_PAY | CURR 13(2) | | Salary |
| BUKRS | CHAR 4 | | Company Code |
| STATUS | CHAR 1 | | A=Active, I=Inactive |
| BEGDA | DATS | | Start Date |
| ENDDA | DATS | | End Date |

**Technical Settings:** Data Class `APPL0`, Size Category `0`

## 3. **Complete ABAP Source Code**

```abap
PROGRAM zmp_employee_sys MESSAGE-ID zemp_msg.

TABLES: zemp_master.

" Global data structures
DATA: ok_code TYPE sy-ucomm,
      save_ok TYPE sy-ucomm,
      gv_pernr TYPE zemp_master-pernr,
      gv_mode TYPE c VALUE 'C',  " C=Create, D=Display
      gt_alv TYPE TABLE OF zemp_master,
      gs_alv TYPE zemp_master.

" BDC structures
DATA: lt_bdcdata TYPE TABLE OF bdcdata,
      ls_bdcdata TYPE bdcdata.

************************************************************************
* MAIN SCREEN 0100 - MENU & SELECTION
************************************************************************
MODULE status_0100 OUTPUT.
  SET PF-STATUS 'MAIN_MENU'.
  SET TITLEBAR 'TITLE_MAIN'.
  MESSAGE s005.  " Welcome message
ENDMODULE.

MODULE user_command_0100 INPUT.
  save_ok = ok_code.
  CLEAR: ok_code.
  
  CASE save_ok.
    WHEN 'CREATE'.
      gv_mode = 'C'.
      SET SCREEN 0200.
    WHEN 'DISPLAY'.
      gv_mode = 'D'.
      SET SCREEN 0300.
    WHEN 'REPORT'.
      PERFORM generate_alv_report.
      SET SCREEN 0400.
    WHEN 'BDC_DEMO'.
      PERFORM bdc_pa40_demo.
    WHEN 'BAPI_DEMO'.
      PERFORM bapi_employee_demo.
    WHEN 'BACK' OR 'EXIT' OR 'CANC'.
      LEAVE PROGRAM.
  ENDCASE.
ENDMODULE.

************************************************************************
* SCREEN 0200 - CREATE EMPLOYEE
************************************************************************
MODULE status_0200 OUTPUT.
  SET PF-STATUS 'EMP_MENU'.
  SET TITLEBAR 'TITLE_CREATE'.
ENDMODULE.

MODULE user_command_0200 INPUT.
  save_ok = ok_code.
  CLEAR: ok_code.
  
  CASE save_ok.
    WHEN 'SAVE'.
      PERFORM validate_employee_data.
      PERFORM save_employee_data.
      SET SCREEN 0100.
    WHEN 'BACK'.
      SET SCREEN 0100.
    WHEN 'CANCEL'.
      CLEAR: gs_alv.
      SET SCREEN 0100.
  ENDCASE.
ENDMODULE.

************************************************************************
* SCREEN 0300 - DISPLAY EMPLOYEE
************************************************************************
MODULE status_0300 OUTPUT.
  SET PF-STATUS 'EMP_MENU'.
  SET TITLEBAR 'TITLE_DISPLAY'.
  PERFORM fetch_employee_data.
ENDMODULE.

MODULE user_command_0300 INPUT.
  save_ok = ok_code.
  CLEAR ok_code.
  
  CASE save_ok.
    WHEN 'BACK'.
      SET SCREEN 0100.
  ENDCASE.
ENDMODULE.

************************************************************************
* SCREEN 0400 - REPORTS & MESSAGES
************************************************************************
MODULE status_0400 OUTPUT.
  SET PF-STATUS 'REPORT_MENU'.
  SET TITLEBAR 'TITLE_REPORT'.
ENDMODULE.

MODULE user_command_0400 INPUT.
  save_ok = ok_code.
  CLEAR ok_code.
  
  CASE save_ok.
    WHEN 'BACK'.
      SET SCREEN 0100.
  ENDCASE.
ENDMODULE.

************************************************************************
* BUSINESS LOGIC FORMS
************************************************************************
FORM validate_employee_data.
  " Validate all input fields
  IF gs_alv-pernr IS INITIAL.
    MESSAGE e002 WITH 'New'.
  ENDIF.
  
  IF gs_alv-emp_name IS INITIAL.
    MESSAGE e002.
  ENDIF.
  
  IF gs_alv-basic_pay < 10000.
    MESSAGE e003.
  ENDIF.
ENDFORM.

FORM save_employee_data.
  " Simulate PA40 save logic
  gs_alv-mandt = sy-mandt.
  gs_alv-status = 'A'.
  gs_alv-begda = sy-datum.
  gs_alv-endda = '99991231'.
  
  " Insert into custom table
  INSERT zemp_master FROM gs_alv.
  IF sy-subrc = 0.
    MESSAGE s001 WITH gs_alv-pernr.
  ELSE.
    MESSAGE e011.
  ENDIF.
ENDFORM.

FORM fetch_employee_data.
  SELECT SINGLE * FROM zemp_master 
    INTO gs_alv
    WHERE pernr = gv_pernr
      AND mandt = sy-mandt.
    
  IF sy-subrc <> 0.
    MESSAGE e004 WITH gv_pernr.
  ENDIF.
ENDFORM.

FORM generate_alv_report.
  SELECT * FROM zemp_master 
    INTO TABLE gt_alv
    WHERE mandt = sy-mandt
      AND status = 'A'.
  
  MESSAGE s010 WITH lines( gt_alv ).
ENDFORM.

FORM bdc_pa40_demo.
  " Demonstrate PA40 BDC recording
  MESSAGE i009.  " BDC mode selected
  
  " Build BDC data for PA40
  PERFORM bdc_dynpro USING 'MP000000' '2000'.
  PERFORM bdc_field  USING 'P0000-BEGDA' sy-datum.
  PERFORM bdc_field  USING 'P0000-ENDDA' '99991231'.
  PERFORM bdc_field  USING 'BDC_OKCODE' '/00'.
  
  CALL TRANSACTION 'PA40' USING lt_bdcdata 
    MODE 'E' UPDATE 'S'.
    
  IF sy-subrc = 0.
    MESSAGE s011.
  ELSE.
    MESSAGE e011.
  ENDIF.
ENDFORM.

FORM bapi_employee_demo.
  DATA: lt_return TYPE TABLE OF bapiret2,
        ls_return TYPE bapiret2.
  
  MESSAGE i009.  " BAPI mode selected
  
  CALL FUNCTION 'BAPI_EMPLOYEE_GETDATA'
    EXPORTING
      employeenumber = gv_pernr
    TABLES
      return         = lt_return.
      
  READ TABLE lt_return INTO ls_return WITH KEY type = 'E'.
  IF sy-subrc <> 0.
    MESSAGE s012 WITH gv_pernr.
  ELSE.
    MESSAGE e004 WITH gv_pernr.
  ENDIF.
ENDFORM.

" BDC Helper routines
FORM bdc_dynpro USING p_program p_dynpro.
  CLEAR ls_bdcdata.
  ls_bdcdata-program  = p_program.
  ls_bdcdata-dynpro   = p_dynpro.
  ls_bdcdata-dynbegin = 'X'.
  APPEND ls_bdcdata TO lt_bdcdata.
ENDFORM.

FORM bdc_field USING p_fnam p_fval.
  CLEAR ls_bdcdata.
  ls_bdcdata-fnam = p_fnam.
  ls_bdcdata-fval = p_fval.
  APPEND ls_bdcdata TO lt_bdcdata.
ENDFORM.
```
